&& \int\limits_0^{\pi/2} e^{\sin(x)}\cos(x) dx \\ \nonumber
&& \int\limits_0^{\sqrt{ln\pi}} 2xe^{x^2}\cos(e^{x^2}) dx  \\ \nonumber
&& \int\limits_1^{4} \frac{dy}{2\sqrt{y}(1+\sqrt{y})^2}  \\ \nonumber
\end{eqnarray}
